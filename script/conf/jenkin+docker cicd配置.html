#!/bin/bash
path=/home/data/jenkins_home/workspace/jwolf-micro/$module_name
echo 当前目录：$PWD
ssh root@192.168.1.18 "docker rm -f $module_name"
ssh root@192.168.1.18 "cd $path && docker build -t $module_name  ."
 
case $module_name in
  "jwolf-auth")
  expose_port=9402
  debug_port=9502
  ;;
  "jwolf-manage")
  expose_port=8888
  debug_port=8988
  ;;
esac  
  
#启动新容器
ssh root@192.168.1.18 "docker run  -d --restart=always --name $module_name -p $expose_port:$expose_port -p $debug_port:$debug_port -v /var/log/jwolf:/var/log/jwolf --net jwolf_jwolf $module_name"
ssh root@192.168.1.18 "docker image prune -f"
